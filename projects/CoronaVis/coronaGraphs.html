def getCountrybyName(countryname = 'india', deltat = 'W'):
    country = df[df['countryname'] == countryname]
    country_grouped = country.groupby(pd.Grouper(key='dateRep', freq=deltat)).sum()#.sort_values(by='DateRep')
    return country_grouped

def getCountrybyCode(countrycode = 'ind', deltat = 'W'):
    country = df[df['countrycode'] == countrycode]
    country_grouped = country.groupby(pd.Grouper(key='dateRep', freq=deltat)).sum()#.sort_values(by='DateRep')
    return country_grouped
    
plt.figure(figsize=(12,8))

for c in ['ind', 'deu', 'ita', 'usa', 'esp', 'chn']:

urlindia = 'https://www.mohfw.gov.in/'
res_ind = requests.get(urlindia)

extract_contents = lambda row: [x.text.replace('\n', '') for x in row] 
SHORT_HEADERS = ['SNo', 'State','Indian-Confirmed', 'Foreign-Confirmed','Cured','Death'] 

soup = bs(res_ind.content, 'html.parser') 
header = extract_contents(soup.tr.find_all('th')) 
  
statss = [] 
all_rows = soup.find_all('tr') 
  
for row in all_rows: 
    stat = extract_contents(row.find_all('td')) 
    if stat: 
        if len(stat) == 5: 
            # last row 
            stat = ['', *stat] 
            statss.append(stat) 
        elif len(stat) == 6: 
            statss.append(stat) 
  
statss[-1][1] = "Total Cases"
  
statss.remove(statss[-1]) 

df_ind = pd.DataFrame(statss, columns=SHORT_HEADERS)
df_ind = df_ind.iloc[1:28]

for c in ['SNo','Indian-Confirmed', 'Foreign-Confirmed', 'Cured', 'Death']:
    df_ind = df_ind.astype({c: int})

df_ind.sort_values(by=['Indian-Confirmed'], ascending=False, inplace=True) #Indian-Confirmed
