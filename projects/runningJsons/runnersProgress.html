class activitiesPolar(object):
    def __init__(self,wdir=fpath, whichactivity="RUNNING", readall=False, lowerPaceThr=9.5):
        #
        #==== List all json files in the given folder ==========
        self.fpath = fpath + r'\\'
        self.allfiles = [ os.path.basename(x) for x in glob.glob(fpath+'\*.json')]
        
        #==== Select only the training session files ===========
        self.allfiles = [v for v in self.allfiles if (v.split('-')[0] == 'training') & (v.split('-')[1] == 'session')]
        self.readActivityFiles()
        
    def readActivityFiles(self):
        activity_summary = array(['start_time', 'activity_type', 'activity_distance', 'activity_duration', 'avg_speed', 'max_speed', 'avg_hrate', 'avg_cadence'])
        
        for filename in self.allfiles:
            with open(self.fpath + filename, 'r') as infile:
                self.data = json.load(infile)
                start_time = datetime.datetime.fromisoformat(self.data['startTime'])
                activity_type = self.data['exercises'][0]['sport']
                activity_distance = self.data['distance']/1e3 #km
                activity_duration = float(self.data['duration'].split("PT")[1].split("S")[0])/3600 #in hours
                avg_speed = self.data['exercises'][0]['speed']['avg']
                max_speed = self.data['exercises'][0]['speed']['max']
                #=============
                if 'cadence' in self.data['exercises'][0].keys():
                    avg_cadence = self.data['exercises'][0]['cadence']['avg']
                else: avg_cadence = float('NaN')
                
                #=============
                if 'heartRate' in self.data['exercises'][0].keys():
                    avg_hrate = self.data['exercises'][0]['heartRate']['avg']
                else: avg_hrate = float('NaN')
                #=============
                activity_array = array([start_time, activity_type, activity_distance, activity_duration, avg_speed, max_speed, avg_hrate, avg_cadence ])
                #print(len(activity_array))
                    
            activity_summary = vstack((activity_summary, activity_array))
            # save a Pandas DF.
        self.activity_summary_df = pd.DataFrame(activity_summary[1:,:], columns=activity_summary[0,:])
        running_mask = self.activity_summary_df['activity_type'] == 'RUNNING'
        cycling_mask = self.activity_summary_df['activity_type'] == 'CYCLING'
        #===
        self.running_df = self.activity_summary_df[running_mask]
        self.cycling_df = self.activity_summary_df[cycling_mask]

    def describeStats(self, activity='running'):
        fig, ax = subplots(2,2, figsize=(6,6))
        plot_list = ['activity_distance', 'activity_duration', 'avg_speed', 'avg_hrate']
        for i,v in enumerate(plot_list):
            if activity == 'running':
                self.running_df[plot_list[i]].hist(bins=12, ax=ax.flatten()[i])
            elif activity == 'cycling':
                self.cycling_df[plot_list[i]].hist(bins=12, ax=ax.flatten()[i])
            ax.flatten()[i].set_title(plot_list[i])

Improvements (or not)¶

Improvement is when, over time value of pace decreases, and the corresponding avg. heart rate also lowers. Below, I create a feature, product of pace and hrate, normalized to 7min/km and heart rate of 140 (arbitrarily chosen normalization). (Pace/7.0 x Hrate/140.0) should decrease over time. Instead we see an increase, indicating no improvement! Perhaps, I should run more consistantly!
